 create table usuarios
(
id_usuario int  identity ,
nombres varchar(100),
apellidos varchar(100),
email varchar(100) unique,
contraseña varchar(300),
telefono varchar(15),
estado int ,
PRIMARY KEY(id_usuario)
);

 create table roles
(
id_rol int   identity ,	
nombre_rol varchar(100),
descripcion varchar(100),
PRIMARY KEY(id_rol)
);

create table usuarios_roles
(
id int identity ,
id_rol int REFERENCES roles(id_rol), 
id_usuario int  REFERENCES usuarios(id_usuario),

PRIMARY KEY(id)
);

create table aulas(
id_aula int identity , 
tipo varchar(15) ,	
capacidad int,
descripcion varchar(100),
PRIMARY KEY (id_aula)
);


create table equipo
(
id_equipo int identity ,
tipo varchar(30)    ,
descripcion varchar(100),
PRIMARY KEY(id_equipo)
);


create table hora(
id_hora int identity ,
hora_entrada varchar(10),
hora_salida varchar(10),
PRIMARY KEY(id_hora)
);



Create table dia_hora_lab
(
id_dia_hora_lab int identity,
fecha varchar(10),
id_aula int REFERENCES aulas(id_aula),
id_hora int REFERENCES hora(id_hora),
PRIMARY KEY(id_dia_hora_lab)
)



create table reservaciones
( 
id int   identity ,    
id_usuario int   ,
id_aula int REFERENCES aulas(id_aula),  
id_hora int REFERENCES hora(id_hora),
fecha varchar(15),
hora varchar(20), 
equipo varchar(100),
estado int,
Primary Key(id)
);
